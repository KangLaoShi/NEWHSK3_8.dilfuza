              <p className="mt-4 text-green-400 text-sm">正确答案 (To'g'ri javob): {q.ans}</p>

            )}

          </div>

        ))}

      </div>


      {!submitted && Object.keys(answers).length === grammarQuestions.length && (

        <div className="mt-10 flex justify-center">

          <button onClick={() => setSubmitted(true)} className="px-10 py-4 bg-gradient-to-r from-indigo-500 to-purple-600 hover:from-indigo-400 hover:to-purple-500 text-white font-bold rounded-full shadow-lg transform hover:scale-105 transition-all">

            Tekshirish (Jo'natish)

        </div>

          <p className="text-2xl text-indigo-200">{getScore()} / {grammarQuestions.length} to'g'ri</p>

        </div>

      )}

    </div>

};



const FillBlanks = () => {

  const [answers, setAnswers] = useState({});

  const [submitted, setSubmitted] = useState(false);


  const handleSelect = (qIndex, slotIndex, value) => {

    if(!submitted) {

      setAnswers(prev => ({

        ...prev, 

        [`${qIndex}-${slotIndex}`]: value

      }));

    }

  };


  const calculateScore = () => {

    let score = 0;

    let total = 0;

    fillBlanksQuestions.forEach((q, qIndex) => {

      const correctAns = Array.isArray(q.ans) ? q.ans : [q.ans];

      correctAns.forEach((ans, slotIndex) => {

        total++;

        if(answers[`${qIndex}-${slotIndex}`] === ans) score++;

      });

    });

    return { score, total };

  };


  const renderQuestion = (q, qIndex) => {

    const parts = q.q.split('【 】');

    const correctAns = Array.isArray(q.ans) ? q.ans : [q.ans];


    return (

      <div className="text-xl text-white leading-loose">

        {parts.map((part, i) => (

          <React.Fragment key={i}>

            <span>{part}</span>

            {i < parts.length - 1 && (

              <span className="mx-2 inline-block">

                <select 

                  className={`bg-indigo-950/80 border text-center appearance-none px-4 py-1 rounded-lg outline-none transition-colors

                    ${submitted && answers[`${qIndex}-${i}`] === correctAns[i] ? 'border-green-400 text-green-300' : ''}

                    ${submitted && answers[`${qIndex}-${i}`] !== correctAns[i] ? 'border-red-400 text-red-300' : ''}

                    ${!submitted ? 'border-indigo-400 text-indigo-200 hover:border-indigo-300' : ''}

                  `}

                  value={answers[`${qIndex}-${i}`] || ""}

                  onChange={(e) => handleSelect(qIndex, i, e.target.value)}

                  disabled={submitted}

                >

                  <option value="" disabled>---</option>

                  {fillBlankWordBank.map(w => (

                     <option key={w} value={w}>{w}</option>

                  ))}

                </select>

              </span>

            )}

          </React.Fragment>

        ))}

        {submitted && (

          <div className="text-sm mt-2 flex gap-4">

            {correctAns.map((ca, i) => answers[`${qIndex}-${i}`] !== ca && (

               <span key={i} className="text-green-400">To'g'ri: {ca}</span>

            ))}

          </div>

        )}

      </div>

    );

  };


  const { score, total } = calculateScore();


  return (

    <div className="max-w-5xl mx-auto">

       <div className="flex justify-between items-center mb-8">

        <h2 className="text-3xl font-bold text-white">So'zlarni to'ldirish (Bo'shliqlarni to'ldiring)</h2>

        <button onClick={() => { setSubmitted(false); setAnswers({}); }} className="flex items-center text-indigo-300 hover:text-white transition">

          <RotateCcw size={18} className="mr-2" /> Qayta boshlash

        </button>

      </div>


      <div className="bg-indigo-900/30 border border-white/10 rounded-2xl p-6 mb-8 backdrop-blur-sm">

        <p className="text-indigo-200 mb-4 font-semibold">Kerakli so'zlar (So'z boyligi):</p>

        <div className="flex flex-wrap gap-3">

          {fillBlankWordBank.map((w, i) => (

             <span key={i} className="bg-white/10 border border-white/20 text-white px-3 py-1 rounded-full text-sm">{w}</span>

          ))}

        </div>

      </div>


      <div className="space-y-6">

        {fillBlanksQuestions.map((q, index) => (

          <div key={index} className="bg-white/5 border border-white/10 rounded-2xl p-6 backdrop-blur-sm">

            {renderQuestion(q, index)}

          </div>

        ))}

      </div>


      {!submitted && Object.keys(answers).length > 0 && (

        <div className="mt-10 flex justify-center">

          <button onClick={() => setSubmitted(true)} className="px-10 py-4 bg-gradient-to-r from-indigo-500 to-purple-600 hover:from-indigo-400 hover:to-purple-500 text-white font-bold rounded-full shadow-lg transform hover:scale-105 transition-all">

            Tekshirish (Jo'natish)

          </button>

        </div>

      )}


      {submitted && (

        <div className="mt-10 p-8 bg-gradient-to-br from-indigo-900/80 to-purple-900/80 rounded-2xl border border-indigo-400/50 text-center shadow-2xl">

          <h3 className="text-4xl font-bold text-white mb-2">Natija</h3>

          <p className="text-2xl text-indigo-200">{score} / {total} to'g'ri javob</p>

        </div>

      )}

    </div>

  );

};
